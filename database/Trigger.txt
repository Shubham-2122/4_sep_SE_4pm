जब नया student insert हो, तो उसकी entry Log table में अपने आप चली जाए

1. Main Table (students)
CREATE TABLE students (
    id INT PRIMARY KEY AUTO_INCREMENT,
    fname VARCHAR(50),
    lname VARCHAR(50),
    email VARCHAR(100),
    mobile VARCHAR(15)
);

2. Log Table (changes track करने के लिए)

CREATE TABLE student_log (
    log_id INT PRIMARY KEY AUTO_INCREMENT,
    action VARCHAR(20),
    fname VARCHAR(50),
    lname VARCHAR(50),
    email VARCHAR(100),
    mobile VARCHAR(15),
    action_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

3. Trigger बनाना (AFTER INSERT)

DELIMITER //
CREATE TRIGGER after_student_insert
AFTER INSERT on student
FOR EACH ROW
BEGIN
	INSERT INTO student_log (ACTION, fname, lanme, email, mobile)
    VALUES ('INSERT',NEW.fname, NEW.lanme, NEW.email, NEW.mobile);
END //

DELIMITER ;
